# for döngüsü
~~~javascript
// for döngüsü
for(var i = 0; i<5 ; i++){
    console.log(i);
}
~~~

# While döngüsü
~~~javascript
//while döngüsü
var j = 0;
while(j<5){
    console.log(j);
    j++;
}
~~~
# do-while döngüsü
~~~javascript
//do-while döngüsü
var k=0;
do{
console.log(k);
k++;
}while(k<5);
~~~
